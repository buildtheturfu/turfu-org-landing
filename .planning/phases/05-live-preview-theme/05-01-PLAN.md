Create SaveIndicator component and useBeforeUnload hook, integrate into ArticleEditor:

1. Create `src/hooks/useBeforeUnload.ts`:
   ```typescript
   import { useEffect } from 'react';

   export function useBeforeUnload(isDirty: boolean) {
     useEffect(() => {
       const handler = (e: BeforeUnloadEvent) => {
         if (isDirty) {
           e.preventDefault();
           return '';
         }
       };

       window.addEventListener('beforeunload', handler);
       return () => window.removeEventListener('beforeunload', handler);
     }, [isDirty]);
   }
   ```

2. Create `src/components/admin/SaveIndicator.tsx`:
   ```typescript
   import { Check, AlertCircle } from 'lucide-react';

   interface SaveIndicatorProps {
     isDirty: boolean;
   }

   export default function SaveIndicator({ isDirty }: SaveIndicatorProps) {
     if (isDirty) {
       return (
         <span className="flex items-center gap-1.5 text-sm text-yellow-500">
           <AlertCircle size={14} />
           Unsaved changes
         </span>
       );
     }

     return (
       <span className="flex items-center gap-1.5 text-sm text-foreground-muted">
         <Check size={14} />
         Saved
       </span>
     );
   }
   ```
